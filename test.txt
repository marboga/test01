//Print all the integers from 1 to 255.

function count() {
    for(var i = 1; i <= 255; i++) {
        console.log(i);
    }
}

count();

//Print Odds 1-255
function count() {
    for(var i = 1; i <= 255; i = i + 2) {
        console.log(i);
    }
}

count();

//Print integers from 0 to 255, and with each integer print the sum so far.
function sum() {
  var sum = 0;
  for (var i = 0; i <= 255; i++) {
    sum = sum + i;
    console.log(i + ' , ' + sum);
  }
}

sum();
